% ---- symlog y-axis transformation ----
% ---- symlog y-axis transformation ----
% Two-step approach: first compute a safe integer flag (no ln involved),
% then branch at TeX level with \ifnum to avoid pgfmath's eager evaluation.
\def\SLT{0.01}% threshold

\makeatletter
\pgfplotsset{
    symlog y/.style={
        y coord trafo/.code={%
            \pgfmathparse{abs(##1) > \SLT ? 1 : 0}%
            \pgfmathtruncatemacro{\SL@flag}{\pgfmathresult}%
            \ifnum\SL@flag=1\relax
                \pgfmathparse{(##1 > 0 ? 1 : -1)*(\SLT + ln(abs(##1)/\SLT))}%
            \else
                \pgfmathparse{##1}%
            \fi
        },
        y coord inv trafo/.code={%
            \pgfmathparse{abs(##1) > \SLT ? 1 : 0}%
            \pgfmathtruncatemacro{\SL@flag}{\pgfmathresult}%
            \ifnum\SL@flag=1\relax
                \pgfmathparse{(##1 > 0 ? 1 : -1)*\SLT*exp(abs(##1) - \SLT)}%
            \else
                \pgfmathparse{##1}%
            \fi
        },
    },
}
\makeatother
